📋 Étape 4 — Règles Firestore (manuel)
Cette étape doit être faite manuellement dans la console Firebase. Copie ces règles dans Firebase Console → Firestore → Règles : 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{uid}/{col}/{doc=**} {
      allow read, write: if request.auth != null
                         && request.auth.uid == uid;
    }
    match /org_new_benef/{doc} {
      allow read, write: if request.auth != null;
    }
    match /audit_logs/{doc} {
      allow read, write: if request.auth != null;
    }
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
